export const createOrderController = async (req: NextRequest) => {
    try {
        const currentUser = await authenticate(req);
        authorize(currentUser, "order.create");

        const body = await req.json();
        const result = createOrderSchema.safeParse(body);

        if (!result.success) {
            return NextResponse.json(
                new ApiResponse(false, "Validation failed", result.error.flatten().fieldErrors),
                { status: 400 }
            );
        }

        const order = await createOrderService(result.data);

        return NextResponse.json(
            new ApiResponse(true, "Order created successfully", order),
            { status: 201 }
        );
    } catch (error) {
        if (error instanceof ApiError) {
            return NextResponse.json(
                new ApiResponse(false, error.message, error.errors),
                { status: error.statusCode }
            );
        }
        return NextResponse.json(new ApiResponse(false, "Internal Server Error"), { status: 500 });
    }
};

export const getOrdersController = async (req: NextRequest) => {
    try {
        const currentUser = await authenticate(req);
        authorize(currentUser, "order.read");

        const { searchParams } = new URL(req.url);

        const result = getOrdersSchema.safeParse({
            page: searchParams.get("page"),
            limit: searchParams.get("limit"),
            customerId: searchParams.get("customerId") ?? undefined,
            outletId: searchParams.get("outletId") ?? undefined,
            orderStatus: searchParams.get("orderStatus") ?? undefined,
            paymentStatus: searchParams.get("paymentStatus") ?? undefined,
        });

        if (!result.success) {
            return NextResponse.json(
                new ApiResponse(false, "Validation failed", result.error.flatten().fieldErrors),
                { status: 400 }
            );
        }

        let { page, limit, customerId, outletId, orderStatus, paymentStatus } = result.data;
        
        let assignedDeliveryStaffId = undefined;

        // Enforce Tenant Isolation for managers
        if (currentUser.accessScope === "outlet") {
            outletId = currentUser.outletId?.toString();
        }
        
        // Enforce Rider Isolation
        if (currentUser.isRider) {
            assignedDeliveryStaffId = currentUser._id.toString();
        }

        const data = await getOrdersService(page, limit, customerId, outletId, orderStatus, paymentStatus, assignedDeliveryStaffId);

        return NextResponse.json(
            new ApiResponse(true, "Orders fetched successfully", data)
        );
    } catch (error) {
        if (error instanceof ApiError) {
            return NextResponse.json(
                new ApiResponse(false, error.message, error.errors),
                { status: error.statusCode }
            );
        }
        return NextResponse.json(new ApiResponse(false, "Internal Server Error"), { status: 500 });
    }
};

export const getOrderByIdController = async (req: NextRequest, id: string) => {
    try {
        const currentUser = await authenticate(req);
        authorize(currentUser, "order.read");

        const result = orderIdSchema.safeParse({ id });

        if (!result.success) {
            return NextResponse.json(
                new ApiResponse(false, "Validation failed", result.error.flatten().fieldErrors),
                { status: 400 }
            );
        }

        const order = await getOrderByIdService(result.data.id);

        if (currentUser.accessScope === "outlet" && order.outletId?._id?.toString() !== currentUser.outletId?.toString()) {
            return NextResponse.json(new ApiResponse(false, "You do not have permission to access data of another outlet"), { status: 403 });
        }
        
        if (currentUser.isRider && order.assignedDeliveryStaffId?._id?.toString() !== currentUser._id.toString()) {
            return NextResponse.json(new ApiResponse(false, "You do not have permission to access this order"), { status: 403 });
        }

        return NextResponse.json(
            new ApiResponse(true, "Order fetched successfully", order)
        );
    } catch (error) {
        if (error instanceof ApiError) {
            return NextResponse.json(new ApiResponse(false, error.message), { status: error.statusCode });
        }
        return NextResponse.json(new ApiResponse(false, "Internal Server Error"), { status: 500 });
    }
};

export const updateOrderController = async (req: NextRequest, id: string) => {
    try {
        const currentUser = await authenticate(req);
        authorize(currentUser, "order.update");

        const idResult = orderIdSchema.safeParse({ id });
        if (!idResult.success) {
            return NextResponse.json(
                new ApiResponse(false, "Invalid Order ID", idResult.error.flatten().fieldErrors),
                { status: 400 }
            );
        }

        const orderCheck = await getOrderByIdService(idResult.data.id);
        if (currentUser.accessScope === "outlet" && orderCheck.outletId?._id?.toString() !== currentUser.outletId?.toString()) {
            return NextResponse.json(new ApiResponse(false, "You do not have permission to access data of another outlet"), { status: 403 });
        }
        
        if (currentUser.isRider && orderCheck.assignedDeliveryStaffId?._id?.toString() !== currentUser._id.toString()) {
            return NextResponse.json(new ApiResponse(false, "You do not have permission to access this order"), { status: 403 });
        }

        const body = await req.json();
        
        // If it's a rider, they can ONLY update orderStatus, deliveryStatus, etc. We should probably restrict them further, 
        // but for now we rely on the schema or UI hiding the fields.
        const result = updateOrderSchema.safeParse(body);

        if (!result.success) {
            return NextResponse.json(
                new ApiResponse(false, "Validation failed", result.error.flatten().fieldErrors),
                { status: 400 }
            );
        }

        const order = await updateOrderService(idResult.data.id, result.data);

        return NextResponse.json(
            new ApiResponse(true, "Order updated successfully", order)
        );
    } catch (error) {
        if (error instanceof ApiError) {
            return NextResponse.json(
                new ApiResponse(false, error.message, error.errors),
                { status: error.statusCode }
            );
        }
        return NextResponse.json(new ApiResponse(false, "Internal Server Error"), { status: 500 });
    }
};

export const getSettingsService = async () => {
    // Since it's a global setting, we usually just have one document.
    let settings = await Settings.findOne();
    
    // If no settings exist yet, create a default one
    if (!settings) {
        settings = await Settings.create({
            restaurantName: "My Restaurant",
            maximumOutletLimit: 9, // As per business rule
        });
    }

    return settings;
};

export const updateSettingsService = async (data: any) => {
    let settings = await Settings.findOne();
    
    if (!settings) {
        // If it doesn't exist, create it with the provided data
        settings = await Settings.create({
            restaurantName: "My Restaurant",
            ...data,
        });
        return settings;
    }

    // Update the existing single document
    const updatedSettings = await Settings.findByIdAndUpdate(
        settings._id,
        { $set: data },
        { new: true, runValidators: true }
    );

    return updatedSettings;
};

Yahan saare **Admin Review APIs** ke endpoints aur Apidog/Postman mein test karne ke liye unke exact JSON payloads hain. 

> **Base URL:** `https://niragoapi.vercel.app/api/v1/admin/reviews` (ya `http://localhost:3000/...` for local)
> **Headers Needed:** `Authorization: Bearer <ADMIN_TOKEN>`

---

### 1. Create a Review (POST)
**Endpoint:** `POST /api/v1/admin/reviews`

*(Note: Isme valid `customerId`, `orderId`, `outletId`, aur `menuItemId` daalna padega, varna 404 aayega kyunki database in relations ko verify karega)*

**JSON Body:**
```json
{
    "customerId": "6a51c0ebe2c6003316af4483",
    "orderId": "6a51c4603ade1b28b43a9ae2",
    "outletId": "6a51c4603ade1b28b43a9ae5",
    "menuItemId": "6a51c4603ade1b28b43a9ae8",
    "rating": 5,
    "comment": "The food was absolutely amazing and the packaging was great!",
    "images": [
        "data:image/png;base64,iVBORw0KGgoAAAANSUhEUgAAAAEAAAABCAQAAAC1HAwCAAAAC0lEQVR42mNkYAAAAAYAAjCB0C8AAAAASUVORK5CYII="
    ],
    "outletResponse": "Thank you so much for your kind words! We hope to serve you again."
}
```

---

### 2. Get All Reviews (GET)
**Endpoint:** `GET /api/v1/admin/reviews`

**Query Params (Optional for filtering/pagination):**
- `?page=1&limit=10`
- `?search=amazing` (comment mein search karega)
- `?rating=5`
- `?outletId=6a51c...`
- `?customerId=6a51c...`
- `?menuItemId=6a51c...`

---

### 3. Get Single Review By ID (GET)
**Endpoint:** `GET /api/v1/admin/reviews/{id}`
*(Replace `{id}` with the Review ID you get from the POST/GET response)*

---

### 4. Update a Review (PATCH)
**Endpoint:** `PATCH /api/v1/admin/reviews/{id}`
*(Replace `{id}` with the actual Review ID)*

**JSON Body:** *(Tum sirf wohi fields bhejo jo update karni hain)*
```json
{
    "rating": 4,
    "comment": "Food was good, but delivery was a bit late.",
    "outletResponse": "We apologize for the delay. We are improving our delivery times!"
}
```
*(Pro-tip: Jab bhi tum `outletResponse` bhejoge aur woh empty string `""` nahi hoga, toh backend automatically `responsePublishedAt` ka timestamp set kar dega!)*

---

### 5. Delete a Review (DELETE)
**Endpoint:** `DELETE /api/v1/admin/reviews/{id}`
*(Replace `{id}` with the actual Review ID)*

**No JSON Body needed.** Sirf ID URL mein paas karni hai.

            page: searchParams.get("page"),

            customerId: searchParams.get("customerId") ?? undefined,
            outletId: searchParams.get("outletId") ?? undefined,
            orderStatus: searchParams.get("orderStatus") ?? undefined,
            paymentStatus: searchParams.get("paymentStatus") ?? undefined,

            );

        if (currentUser.isRider) {
            assignedDeliveryStaffId = currentUser._id.toString();
        }

        return NextResponse.json(
            new ApiResponse(true, "Orders fetched successfully", data)
        );

        if (error instanceof ApiError) {
            return NextResponse.json(
                new ApiResponse(false, error.message, error.errors),

import Order from "@/models/Order";
import Customer from "@/models/Customer";
import Outlet from "@/models/Outlet";
import { ApiError } from "@/utils/ApiError";
import mongoose from "mongoose";

export const createOrderService = async (data: any) => {
    // Check order number uniqueness
    const existingOrder = await Order.findOne({ orderNumber: data.orderNumber });
    if (existingOrder) {
        throw new ApiError(409, "Order with this order number already exists");
    }

    const customer = await Customer.findById(data.customerId);
    if (!customer) {
        throw new ApiError(404, "Customer not found");
    }

    const outlet = await Outlet.findById(data.outletId);
    if (!outlet) {
        throw new ApiError(404, "Outlet not found");
    }

    const order = await Order.create(data);
    return order;
};

export const getOrdersService = async (
    page: number,
    limit: number,
    customerId?: string,
    outletId?: string,
    orderStatus?: string,
    paymentStatus?: string,
    assignedDeliveryStaffId?: string
) => {
    const skip = (page - 1) * limit;
    const query: any = {};
    
    if (customerId) query.customerId = customerId;
    if (outletId) query.outletId = outletId;
    if (orderStatus) query.orderStatus = orderStatus;
    if (paymentStatus) query.paymentStatus = paymentStatus;
    if (assignedDeliveryStaffId) query.assignedDeliveryStaffId = assignedDeliveryStaffId;

    const [orders, total] = await Promise.all([
        Order.find(query)
            .populate("customerId", "name phone email")
            .populate("outletId", "name code")
            .populate("assignedDeliveryStaffId", "name phone")
            .sort({ createdAt: -1 })
            .skip(skip)
            .limit(limit),
        Order.countDocuments(query),
    ]);

    return {
        orders,
        pagination: {
            total,
            page,
            limit,
            totalPages: Math.ceil(total / limit),
        },
    };
};

export const getOrderByIdService = async (id: string) => {
    if (!mongoose.Types.ObjectId.isValid(id)) {
        throw new ApiError(400, "Invalid order id");
    }

    const order = await Order.findById(id)
        .populate("customerId", "name phone email")
        .populate("outletId", "name code")
        .populate("assignedDeliveryStaffId", "name phone vehicleNumber")
        .populate("items.menuItemId", "name price image isVeg");
    
    if (!order) {
        throw new ApiError(404, "Order not found");
    }
    
    return order;
};

export const updateOrderService = async (id: string, data: any) => {
    if (!mongoose.Types.ObjectId.isValid(id)) {
        throw new ApiError(400, "Invalid order id");
    }

    const order = await Order.findById(id);
    if (!order) {
        throw new ApiError(404, "Order not found");
    }

    if (data.orderNumber && data.orderNumber !== order.orderNumber) {
        const existingOrder = await Order.findOne({ orderNumber: data.orderNumber, _id: { $ne: id } });
        if (existingOrder) {
            throw new ApiError(409, "Order with this order number already exists");
        }
    }

    if (data.customerId && data.customerId !== order.customerId.toString()) {
        const customer = await Customer.findById(data.customerId);
        if (!customer) throw new ApiError(404, "Customer not found");
    }

    if (data.outletId && data.outletId !== order.outletId.toString()) {
        const outlet = await Outlet.findById(data.outletId);
        if (!outlet) throw new ApiError(404, "Outlet not found");
    }

    const updatedOrder = await Order.findByIdAndUpdate(
        id,
        { $set: data },
        { new: true, runValidators: true }
    )
        .populate("customerId", "name phone email")
        .populate("outletId", "name code")
        .populate("assignedDeliveryStaffId", "name phone vehicleNumber");

    return updatedOrder;
};

export const deleteOrderService = async (id: string) => {
    if (!mongoose.Types.ObjectId.isValid(id)) {
        throw new ApiError(400, "Invalid order id");
    }

    const order = await Order.findById(id);
    if (!order) {
        throw new ApiError(404, "Order not found");
    }

    await Order.findByIdAndDelete(id);
    return order;
};
